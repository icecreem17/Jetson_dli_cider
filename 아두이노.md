#include <cm1106_i2c.h>

CM1106_I2C cm1106_i2c;

void setup() {
  cm1106_i2c.begin();
  Serial.begin(9600);
  delay(1000);
}

void loop() {
  uint8_t ret = cm1106_i2c.measure_result();

  if (ret == 0) {
    Serial.print("CO2:");
    Serial.println(cm1106_i2c.co2); 
    Serial.println("Operating normal");// CO2 데이터를 시리얼로 전송
  } else {
    Serial.println("Error reading sensor data.");
  }
  delay(1000);
}
